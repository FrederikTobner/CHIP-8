46#define GEN_DYNAMIC_ARRAY_TYPE(TYPE) \
47 typedef struct { \
48 size_t allocated; \
49 size_t used; \
50 TYPE * data; \
51 } TYPE##_dynamic_array_t;
52
53#define GEN_DYNAMIC_ARRAY_FUNCTIONS(TYPE) \
54 bool TYPE##_dynamic_array_init(TYPE##_dynamic_array_t * array, size_t init_size, size_t init_used) { \
55 assert(init_size >= init_used); \
56 init_size = MAX(init_size, MIN_ARRAY_SIZE); \
57 array->data = CHECKED_MALLOC_USING_TYPE(init_size, *array->data); \
58 array->allocated = (array->data) ? init_size : 0; \
59 array->used = (array->data) ? init_used : 0; \
60 return !!array->data; \
61 } \
62 \
63 void TYPE##_dynamic_array_free(TYPE##_dynamic_array_t * array) { \
64 free(array->data); \
65 array->data = NULL; \
66 array->allocated = array->used = 0; \
67 } \
68 \
69 bool TYPE##_dynamic_array_resize(TYPE##_dynamic_array_t * array, size_t new_size) { \
70 size_t alloc_size = MAX(new_size, MIN_ARRAY_SIZE); \
71 TYPE * new_data = CHECKED_REALLOC_USING_SIZE(array->data, alloc_size, *array->data); \
72 if (!new_data) \
73 return false; \
74 array->data = new_data; \
75 array->allocated = alloc_size; \
76 array->used = MIN(array->used, new_size); \
77 return true; \
78 } \
79 \
80 bool TYPE##_dynamic_array_write(TYPE##_dynamic_array_t * array, TYPE val) { \
81 if (array->used == array->allocated) { \
82 if (AT_MAX_LENGTH_USING_TYPE(array->allocated, *array->data)) \
83 return false; \
84 size_t new_size = CAPPED_DPL_USING_SIZE(array->allocated, *array->data); \
85 int resize_success = TYPE##_dynamic_array_resize(array, new_size); \
86 if (!resize_success) \
87 return false; \
88 } \
89 array->data[array->used++] = val; \
90 return true; \
91 }
